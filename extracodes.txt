# Print prime numbers
		public static boolean isPrime(int n) {
			if (n <= 1) {
				return false;
			}
			int i = 2;
			while (i * i <= n) {
				if (n % i == 0) {
					return false;
				}
				i++;
			}
			return i * i > n;

		}
		
# Print armstrong number
		public static boolean isArmstrong(int n) {
			int x = n;
			int sum = 0;

			while (n > 0) {
				int m = n % 10;
				n = n / 10;
				sum += (m * m * m);
			}
			return (sum == x);
		}
