1. Reverse an int[]

public static int[] reverseAnArray(int[] arr) {
		int start = 0;
		int end = arr.length-1;
		
		while(start < end) {
			int temp = arr[start];
			arr[start] = arr[end];
			arr[end] = temp;
			start++;
			end--;
	
		}
		
		return arr;
	}


2. Find the minimum number form int[]

	public static int findMinimum(int[] arr) {
		int minimum = arr[0];

		for (int i = 1; i < arr.length; i++) {
			if (arr[i] < minimum)
				minimum = arr[i];

		}
		return minimum;
